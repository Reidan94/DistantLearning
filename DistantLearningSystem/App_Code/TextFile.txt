        public static explicit operator UserModel(Student student) 
        {
            return new UserModel() 
            {
                Id = student.Id,
                Login = student.Login,
                Name = student.Name,
                Password = student.Password,
                UserType = UserType.Student,
                LastName = student.LastName,
                Email = student.Email
            };
        }
